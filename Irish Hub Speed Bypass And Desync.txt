local function makeSlider(yPos, labelText, min, max, default, formatFn, onChange)
	local container = Instance.new("Frame")
	container.Size = UDim2.new(1, -20, 0, 30)
	container.Position = UDim2.new(0, 10, 0, yPos)
	container.BackgroundTransparency = 1
	container.Parent = main

	local lbl = Instance.new("TextLabel")
	lbl.Size = UDim2.new(1, 0, 0, 14)
	lbl.Position = UDim2.new(0, 0, 0, 0)
	lbl.BackgroundTransparency = 1
	lbl.Text = formatFn(default)
	lbl.Font = Enum.Font.GothamBold
	lbl.TextSize = 10
	lbl.TextColor3 = Color3.new(1, 1, 1)
	lbl.TextTransparency = 0.3
	lbl.TextXAlignment = Enum.TextXAlignment.Left
	lbl.Parent = container

	local bar = Instance.new("Frame")
	bar.Size = UDim2.new(1, 0, 0, 5)
	bar.Position = UDim2.new(0, 0, 0, 18)
	bar.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
	bar.BorderSizePixel = 0
	bar.Parent = container
	Instance.new("UICorner", bar).CornerRadius = UDim.new(1, 0)

	local startRel = (default - min) / (max - min)

	local fill = Instance.new("Frame")
	fill.Size = UDim2.new(startRel, 0, 1, 0)
	fill.BackgroundColor3 = COL_GREEN
	fill.BorderSizePixel = 0
	fill.Parent = bar
	Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

	local knobS = Instance.new("Frame")
	knobS.Size = UDim2.new(0, 10, 0, 10)
	knobS.AnchorPoint = Vector2.new(0.5, 0.5)
	knobS.Position = UDim2.new(startRel, 0, 0.5, 0)
	knobS.BackgroundColor3 = Color3.new(1, 1, 1)
	knobS.BorderSizePixel = 0
	knobS.Parent = bar
	Instance.new("UICorner", knobS).CornerRadius = UDim.new(1, 0)

	local thisSliderDragging = false

	bar.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			thisSliderDragging = true
			sliderDragging = true
		end
	end)

	knobS.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			thisSliderDragging = true
			sliderDragging = true
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if thisSliderDragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			local rel = math.clamp((input.Position.X - bar.AbsolutePosition.X) / bar.AbsoluteSize.X, 0, 1)
			rel = math.floor(rel * 180) / 180
			fill.Size = UDim2.new(rel, 0, 1, 0)
			knobS.Position = UDim2.new(rel, 0, 0.5, 0)
			local val = min + rel * (max - min)
			lbl.Text = formatFn(val)
			onChange(val)
		end
	end)

	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			if thisSliderDragging then
				thisSliderDragging = false
				sliderDragging = false
			end
		end
	end)

	return container
end

local function applyDesyncHubFFlags()
	local now = tick()
	local last = (getgenv and getgenv()._desyncFFLast) or 0
	if (now - last) < 0.75 then return end
	if getgenv then getgenv()._desyncFFLast = now end
	if type(setfflag) ~= "function" then return end
	local flags = {
		S2PhysicsSenderRate = 15000,
		DFIntConnectionMTUSize = 1400,
		DFIntRakNetResendBufferArrayLength = 128,
		DFIntRakNetResendTimeoutMS = 300,
		DFIntNetworkLatencyTolerance = 1,
		DFIntNetworkPrediction = 1,
		FFlagRakNetDisableFlowControl = true,
		DFFlagRakNetDisableCongestionControl = true,
		DFIntTaskSchedulerTargetFps = 29383,
		FFlagGameBasicSettingsFramerateCap5 = false,
		FFlagTaskSchedulerLimitTargetFpsTo2402 = false,
	}
	for name, value in pairs(flags) do
		pcall(function() setfflag(tostring(name), tostring(value)) end)
	end
end

local function setRaknetDesync(on)
	pcall(function()
		local g = (getgenv and getgenv()) or _G
		local r = rawget(g, "raknet")
		if type(r) == "table" and type(r.desync) == "function" then r.desync(on) end
		local n = rawget(g, "network")
		if type(n) == "table" and type(n.desync) == "function" then n.desync(on) end
		local f = rawget(g, "fluxus")
		if type(f) == "table" and type(f.network_desync) == "function" then f.network_desync(on) end
	end)
end

local function setupESPForCharacter(character)
	if not desyncEnabled then return end
	local hrp = character:WaitForChild("HumanoidRootPart")
	espLastPosition = hrp.Position
	espSkipFrames = 5
	updateESPPart(espLastPosition, "Server Pos")
	
	if espPreSimConn then espPreSimConn:Disconnect() end
	espPreSimConn = RunService.PreSimulation:Connect(function()
		if not desyncEnabled then return end
		if not character.Parent then
			if espPreSimConn then espPreSimConn:Disconnect() end
			return
		end
		local currentPos = hrp.Position
		if espSkipFrames > 0 then
			espSkipFrames = espSkipFrames - 1
			espLastPosition = currentPos
			return
		end
		local jumpDist = (currentPos - espLastPosition).Magnitude
		if jumpDist > 3 then
			updateESPPart(currentPos, "Server Pos")
		end
		espLastPosition = currentPos
	end)
end

local function stopESP()
	if espPreSimConn then espPreSimConn:Disconnect() end
	if espCharAddedConn then espCharAddedConn:Disconnect() end
	if espPart then espPart:Destroy() end
	espPreSimConn = nil
	espCharAddedConn = nil
	espPart = nil
	espLastPosition = nil
end

local function startESP()
	if not desyncEnabled then return end
	stopESP()
	if lp.Character then setupESPForCharacter(lp.Character) end
	espCharAddedConn = lp.CharacterAdded:Connect(function(char)
		if desyncEnabled then setupESPForCharacter(char) end
	end)
end

local function updateDesyncMovement(dt)
	dt = (type(dt) == "number" and dt > 0 and dt < 0.5) and dt or (1 / 60)
	local blend = math.clamp(dt * DESYNC_VEL_SMOOTH_HZ, 0.08, 0.5)
	local function applySmoothedHorizontal(targetVx, targetVz)
		local cur = desyncInvisiblePart.AssemblyLinearVelocity
		local vy = cur.Y
		local nvx = cur.X + (targetVx - cur.X) * blend
		local nvz = cur.Z + (targetVz - cur.Z) * blend
		local v = Vector3.new(nvx, vy, nvz)
		if desyncLinearVelocity and desyncLinearVelocity.Parent then
			desyncLinearVelocity.VectorVelocity = v
		else
			desyncInvisiblePart.AssemblyLinearVelocity = v
		end
	end
	if not desyncEnabled or not desyncInvisiblePart or not desyncInvisiblePart.Parent then return end
	local char = lp.Character
	if not char or desyncInvisiblePart.Parent ~= char then return end
	local humanoid = char:FindFirstChildOfClass("Humanoid")
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not humanoid or not hrp then return end
	if humanoid.Health <= 0 then
		pcall(function() if desyncOriginalWalk ~= nil then humanoid.WalkSpeed = desyncOriginalWalk end end)
		desyncLastWalkWritten = nil
		local vy0 = desyncInvisiblePart.AssemblyLinearVelocity.Y
		local zv = Vector3.new(0, vy0, 0)
		if desyncLinearVelocity and desyncLinearVelocity.Parent then
			desyncLinearVelocity.VectorVelocity = zv
		else
			desyncInvisiblePart.AssemblyLinearVelocity = zv
		end
		return
	end
	local moveDir = humanoid.MoveDirection
	local moving = moveDir.Magnitude > 0.02
	local driveUnit = moving and moveDir.Unit or nil
	local function getDesyncSplitWalkAndDrive()
		local orig = desyncOriginalWalk or 16
		orig = math.clamp(orig, 6, 34)
		local t = math.clamp((orig - 6) / 28, 0, 1)
		local driveS = DESYNC_DRIVE_MIN + t * (DESYNC_DRIVE_MAX - DESYNC_DRIVE_MIN)
		local reportW
		if IS_TOUCH_DEVICE then
			reportW = driveS * DESYNC_REPORT_PER_DRIVE_TOUCH + DESYNC_REPORT_PER_DRIVE_OFF_TOUCH
			reportW = math.clamp(reportW, DESYNC_REPORT_WALK_MIN_TOUCH, DESYNC_REPORT_WALK_MAX_TOUCH)
		else
			reportW = driveS * DESYNC_REPORT_PER_DRIVE_PC + DESYNC_REPORT_PER_DRIVE_OFF_PC
			reportW = math.clamp(reportW, DESYNC_REPORT_WALK_MIN, DESYNC_REPORT_WALK_MAX)
		end
		return reportW, driveS
	end
	local reportW, driveS = getDesyncSplitWalkAndDrive()
	if moving then
		if desyncLastWalkWritten == nil or math.abs(desyncLastWalkWritten - reportW) > 0.28 then
			pcall(function() humanoid.WalkSpeed = reportW end)
			desyncLastWalkWritten = reportW
		end
		applySmoothedHorizontal(driveUnit.X * driveS, driveUnit.Z * driveS)
	else
		if desyncLastWalkWritten == nil or math.abs(desyncLastWalkWritten - reportW) > 0.35 then
			pcall(function() humanoid.WalkSpeed = reportW end)
			desyncLastWalkWritten = reportW
		end
		applySmoothedHorizontal(0, 0)
	end
end

local function snapDesyncFollower()
	if not desyncEnabled or not desyncInvisiblePart or not desyncInvisiblePart.Parent then return end
	local c = lp.Character
	if not c or desyncInvisiblePart.Parent ~= c then return end
	local hrp = c:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	local now = tick()
	if desyncLastSnapTime > 0 and (now - desyncLastSnapTime) < DESYNC_SNAP_MIN_INTERVAL then return end
	desyncLastSnapTime = now
	desyncInvisiblePart.CFrame = hrp.CFrame * CFrame.new(0, 0, -2.85)
end

local function teardownDesync()
	if desyncConn then desyncConn:Disconnect() desyncConn = nil end
	if desyncNetworkRefreshConn then desyncNetworkRefreshConn:Disconnect() desyncNetworkRefreshConn = nil end
	if desyncInvisiblePart then pcall(function() desyncInvisiblePart:Destroy() end) end
	desyncInvisiblePart = nil
	for _, inst in ipairs(workspace:GetChildren()) do
		if inst.Name == "DarkHub_DesyncFollower" or inst.Name == "DarkHub_DesyncPos" then
			pcall(function() inst:Destroy() end)
		end
	end
	local char = lp.Character
	local hum = char and char:FindFirstChildOfClass("Humanoid")
	if hum and desyncOriginalWalk ~= nil then
		pcall(function() hum.WalkSpeed = desyncOriginalWalk end)
	end
	desyncOriginalWalk = nil
	desyncLastWalkWritten = nil
	desyncLastSnapTime = 0
	desyncPostSimLastT = 0
	desyncLinearVelocity = nil
	setRaknetDesync(false)
end

local function setupDesyncCharacter(char)
	teardownDesync()
	if not desyncEnabled then return end
	local humanoid = char:WaitForChild("Humanoid", 15)
	local hrp = char:WaitForChild("HumanoidRootPart", 15)
	if not humanoid or not hrp then return end
	desyncOriginalWalk = humanoid.WalkSpeed
	desyncLastWalkWritten = nil
	desyncInvisiblePart = Instance.new("Part")
	desyncInvisiblePart.Name = "DarkHub_DesyncFollower"
	desyncInvisiblePart.Size = Vector3.new(2, 1, 2)
	desyncInvisiblePart.Transparency = 1
	desyncInvisiblePart.CanCollide = false
	desyncInvisiblePart.Anchored = false
	desyncInvisiblePart.Massless = true
	desyncInvisiblePart.Parent = char
	local dhAtt = Instance.new("Attachment")
	dhAtt.Name = "DH_DesyncDrive"
	dhAtt.Parent = desyncInvisiblePart
	local lv = Instance.new("LinearVelocity")
	lv.Name = "DH_DesyncLinearVelocity"
	lv.Attachment0 = dhAtt
	lv.RelativeTo = Enum.ActivationRelativeTo.World
	lv.MaxForce = DesyncTuning.lvMaxForce
	lv.VectorVelocity = Vector3.zero
	lv.Parent = desyncInvisiblePart
	desyncLinearVelocity = lv
	local welded = false
	pcall(function()
		local w = Instance.new("Weld")
		w.Name = "DH_DesyncWeld"
		w.Part0 = hrp
		w.Part1 = desyncInvisiblePart
		w.C0 = CFrame.new(0, 0, -3.85)
		w.Parent = desyncInvisiblePart
		welded = true
	end)
	if not welded then
		desyncInvisiblePart.CFrame = hrp.CFrame * CFrame.new(0, 0, -3.85)
		local wc = Instance.new("WeldConstraint")
		wc.Name = "DH_DesyncWeldConstraint"
		wc.Part0 = hrp
		wc.Part1 = desyncInvisiblePart
		wc.Parent = desyncInvisiblePart
	end
	task.defer(function()
		desyncLastSnapTime = 0
		snapDesyncFollower()
		local ps = RunService.PostSimulation
		if ps then ps:Wait() else RunService.Heartbeat:Wait() end
		snapDesyncFollower()
	end)
	task.delay(0.15, snapDesyncFollower)
	desyncPostSimLastT = os.clock()
	local stepper = RunService.PostSimulation or RunService.Heartbeat
	desyncConn = stepper:Connect(function()
		local now = os.clock()
		local dt = math.clamp(now - desyncPostSimLastT, 1 / 240, 1 / 25)
		desyncPostSimLastT = now
		updateDesyncMovement(dt)
	end)
	local refreshAccum = 0
	desyncNetworkRefreshConn = RunService.Heartbeat:Connect(function(dt)
		if not desyncEnabled then return end
		refreshAccum = refreshAccum + (typeof(dt) == "number" and dt or (1 / 60))
		if refreshAccum < 1.2 then return end
		refreshAccum = 0
		applyDesyncHubFFlags()
	end)
	applyDesyncHubFFlags()
	setRaknetDesync(true)
end

local function enableDesync()
	if desyncEnabled then return end
	desyncEnabled = true
	if lp.Character then setupDesyncCharacter(lp.Character) end
	lp.CharacterAdded:Connect(function(char)
		if desyncEnabled then setupDesyncCharacter(char) end
	end)
	startESP()
end

	if not desyncEnabled then return end
	desyncEnabled = false
	teardownDesync()
	stopESP()
end

do
	local dragging = false
	local dragStart = nil
	local startPos = nil
	main.InputBegan:Connect(function(input)
		if sliderDragging then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = main.Position
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if sliderDragging then dragging = false return end
		if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			local delta = input.Position - dragStart
			local viewport = workspace.CurrentCamera.ViewportSize
			local newX = math.clamp(startPos.X.Offset + delta.X, 0, viewport.X - main.AbsoluteSize.X)
			local newY = math.clamp(startPos.Y.Offset + delta.Y, 0, viewport.Y - main.AbsoluteSize.Y)
			main.Position = UDim2.new(0, newX, 0, newY)
		end
	end)
	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)
end

RunService.RenderStepped:Connect(function()
	local offset = Vector2.new(math.sin(tick() * 1.5), 0)
	for _, g in ipairs(allStrokeGrads) do
		g.Offset = offset
	end
end)

RunService.RenderStepped:Connect(function()
	if speedEnabled then
		local char = lp.Character
		if char then
			local hrp = char:FindFirstChild("HumanoidRootPart")
			local hum = char:FindFirstChildOfClass("Humanoid")
			if hrp and hum and hum.MoveDirection.Magnitude > 0 then
				local dir = hum.MoveDirection.Unit
				local useSpeed = (hum.WalkSpeed <= 25) and stealAmount or speedAmount
				hrp.AssemblyLinearVelocity = Vector3.new(
					dir.X * useSpeed,
					hrp.AssemblyLinearVelocity.Y,
					dir.Z * useSpeed
				)
			end
		end
	end
end)

	if lagEnabled and lagAmount > 0 then
		local t = tick()
		while tick() - t < lagAmount do end
	end
end)
