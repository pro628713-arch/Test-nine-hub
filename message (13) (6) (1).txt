-- ContextActionService bindings for controller
local function setupControllerBindings()

    ContextActionService:UnbindAllActions()
    
    if not Enabled.ControllerMode then return end
    
    -- Auto Left (DPad Left)
    ContextActionService:BindAction("ControllerAutoLeft", function(actionName, inputState, inputObject)
        if inputState == Enum.UserInputState.Begin then
            Enabled.AutoWalkEnabled = not Enabled.AutoWalkEnabled
            if Enabled.AutoWalkEnabled then
                Enabled.AutoRightEnabled = false
                Enabled.BatAimbot = false
                setVisual("AutoRightEnabled", false)
                setVisual("BatAimbot", false)
                
                if not Enabled.SpeedBoost then
                    Enabled.SpeedBoost = true
                    startSpeedBoost()
                    setVisual("SpeedBoost", true)
                end

                stopAutoRight()
                stopBatAimbot()
                startAutoWalk()
            else
                stopAutoWalk()
            end
            setVisual("AutoWalkEnabled", Enabled.AutoWalkEnabled)
            playSound("rbxassetid://6895079813", 0.4, 1)
        end
        return Enum.ContextActionResult.Sink
    end, false, Enum.KeyCode.DPadLeft)
    
    -- Auto Right (DPad Right)
    ContextActionService:BindAction("ControllerAutoRight", function(actionName, inputState, inputObject)
        if inputState == Enum.UserInputState.Begin then
            Enabled.AutoRightEnabled = not Enabled.AutoRightEnabled
            if Enabled.AutoRightEnabled then
                Enabled.AutoWalkEnabled = false
                Enabled.BatAimbot = false

                stopBatAimbot()
                startAutoRight()

                stopAutoRight()
            end
            setVisual("AutoRightEnabled", Enabled.AutoRightEnabled)
            playSound("rbxassetid://6895079813", 0.4, 1)

        return Enum.ContextActionResult.Sink

local function SaveConfig()
    local data = {}
    for k, v in pairs(Enabled) do data[k] = v end
    for k, v in pairs(Values) do if k ~= "ButtonPositions" then data[k] = v end end
    
    local currentPositions = {}
    for name, frame in pairs(DraggableFrames) do
        if frame and frame.Parent then
            currentPositions[name] = {frame.Position.X.Scale, frame.Position.X.Offset, frame.Position.Y.Scale, frame.Position.Y.Offset}
        end
    end
    data.ButtonPositions = currentPositions
    
    data.KEY_SPEED = KEYBINDS.SPEED.Name
    data.KEY_BATAIMBOT = KEYBINDS.BATAIMBOT.Name
    data.KEY_AUTOLEFT = KEYBINDS.AUTOLEFT.Name
    data.KEY_AUTORIGHT = KEYBINDS.AUTORIGHT.Name
    data.KEY_TPDOWN = KEYBINDS.TPDOWN.Name
    data.KEY_HOLDJUMP = KEYBINDS.HOLDJUMP.Name
    
    if writefile then
        pcall(function() writefile(ConfigFileName, HttpService:JSONEncode(data)) end)
        return true
    end
    return false
end

local function getFluctuatingSpeed()
    local minS = Values.MinBoostSpeed
    local maxS = Values.MaxBoostSpeed
    if minS > maxS then minS, maxS = maxS, minS end
    local wave = (math.sin(tick() * 8) + 1) / 2
    local noise = math.random(-50, 50) / 1000
    return minS + ((maxS - minS) * wave) + noise
end

local function findBat()
    local char = Player.Character
    if not char then return nil end
    
    for _, item in ipairs(char:GetChildren()) do
        if item:IsA("Tool") and item.Name:lower():find("bat") then return item end
    end
    
    local bp = Player:FindFirstChild("Backpack")
    if bp then
        for _, item in ipairs(bp:GetChildren()) do
            if item:IsA("Tool") and item.Name:lower():find("bat") then return item end
        end
    end
    
    for _, i in ipairs(SlapList) do local t = char:FindFirstChild(i[2]) or (bp and bp:FindFirstChild(i[2])); if t then return t end end
    return nil
end

local function getClosest(myRoot)
    local target = nil
    local dist = math.huge
    local part = nil
    
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= Player and p.Character then
            local root = p.Character:FindFirstChild("HumanoidRootPart")
            local hum = p.Character:FindFirstChildOfClass("Humanoid")
            
            if root and hum and hum.Health > 0 then
                local d = (root.Position - myRoot.Position).Magnitude
                if d < dist then
                    dist = d
                    target = root
                    part = p.Character:FindFirstChild("UpperTorso") or p.Character:FindFirstChild("Torso") or root
                end
            end
        end
    end
    return target, dist, part
end

local function startBatAimbot()
    if Connections.batAimbot then return end
    breakStealSpeedLock()
    
    Connections.batAimbot = RunService.Heartbeat:Connect(function(dt)
        if not Enabled.BatAimbot then return end
        local char = Player.Character
        if not char then return end
        
        local root = char:FindFirstChild("HumanoidRootPart")
        local hum = char:FindFirstChildOfClass("Humanoid")
        if not root or not hum then return end
        
        local bat = findBat()
        if bat and bat.Parent ~= char then
            hum:EquipTool(bat)
        end
        
        local tRoot, d, tTorso = getClosest(root)
        
        if tRoot and tTorso then
            hum.AutoRotate = false
            
            local targetDir = Vector3.new(tTorso.Position.X - root.Position.X, 0, tTorso.Position.Z - root.Position.Z)
            if targetDir.Magnitude > 0.1 then
                targetDir = targetDir.Unit
                local currentDir = root.CFrame.LookVector
                local currentDirFlat = Vector3.new(currentDir.X, 0, currentDir.Z)
                
                if currentDirFlat.Magnitude > 0.1 then
                    currentDirFlat = currentDirFlat.Unit
                    local crossY = currentDirFlat:Cross(targetDir).Y
                    local dot = math.clamp(currentDirFlat:Dot(targetDir), -1, 1)
                    local angle = math.atan2(crossY, dot)
                    
                    root.AssemblyAngularVelocity = Vector3.new(0, angle * 25, 0)
                end
            end

            if not bypassPart or bypassPart.Parent ~= char then createProxy() end

            local dir = (tTorso.Position - root.Position).Unit
            local speed = Values.AimbotSpeed
            
            if d > 2 then
                bypassPart.AssemblyLinearVelocity = dir * speed
            else
                bypassPart.AssemblyLinearVelocity = tRoot.AssemblyLinearVelocity
            end
        else
            hum.AutoRotate = true
        end
    end)
end

local function stopBatAimbot()
    if Connections.batAimbot then
        Connections.batAimbot:Disconnect()
        Connections.batAimbot = nil
    end
    local char = Player.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then hum.AutoRotate = true end
    end
    cleanupBypass()
end

-- Drop feature
local _wfConns = {}

local function startDrop()
    if Enabled.DropEnabled then
        local colConn = RunService.Stepped:Connect(function()
            if not Enabled.DropEnabled then return end
            for _, p in ipairs(Players:GetPlayers()) do
                if p ~= Player and p.Character then
                    for _, part in ipairs(p.Character:GetChildren()) do
                        if part:IsA("BasePart") then
                            part.CanCollide = false
                        end
                    end
                end
            end
        end)
        table.insert(_wfConns, colConn)

        task.spawn(function()
            while Enabled.DropEnabled do
                RunService.Heartbeat:Wait()
                local c = Player.Character
                local root = c and c:FindFirstChild("HumanoidRootPart")
                
                if not root then continue end
                
                local vel = root.Velocity
                root.Velocity = vel * 10000 + Vector3.new(0, 10000, 0)
                
                RunService.RenderStepped:Wait()
                if root and root.Parent then root.Velocity = vel end
                
                RunService.Stepped:Wait()
                if root and root.Parent then
                    root.Velocity = vel + Vector3.new(0, 0.1, 0)
                end
            end
        end)
    end
end

local function stopDrop()
    for _, c in ipairs(_wfConns) do
        if typeof(c) == "RBXScriptConnection" then c:Disconnect() end
    end
    _wfConns = {}
end

-- Speed Boost (accurate velocity)
local function startSpeedBoost()
    if Connections.speed then return end
    
    createProxy()
    
    Connections.speed = RunService.Heartbeat:Connect(function()
        if not Enabled.SpeedBoost then return end
        if Enabled.BatAimbot or Enabled.AutoWalkEnabled or Enabled.AutoRightEnabled then return end
        if isStealSpeedLocked then return end 
        
        local char = Player.Character
        local hum = char and char:FindFirstChildOfClass("Humanoid")
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        
        if not char or not hum or not hrp then return end
        if not bypassPart or bypassPart.Parent ~= char then createProxy() end
        
        local moveDir = hum.MoveDirection
        if moveDir.Magnitude == 0 then
            bypassPart.AssemblyLinearVelocity = Vector3.new(0, bypassPart.AssemblyLinearVelocity.Y, 0)
            return
        end

        local dynamicSpeed = getFluctuatingSpeed()
        local desiredVel = moveDir.Unit * dynamicSpeed
        bypassPart.AssemblyLinearVelocity = Vector3.new(desiredVel.X, bypassPart.AssemblyLinearVelocity.Y, desiredVel.Z)
    end)
end

local function stopSpeedBoost()
    if Connections.speed then Connections.speed:Disconnect() Connections.speed = nil end
    cleanupBypass()
end

    local plots = workspace:FindFirstChild("Plots"); if not plots then return false end

local function findNearestPrompt()
    local h = Player.Character and Player.Character:FindFirstChild("HumanoidRootPart"); if not h then return nil end
    local plots = workspace:FindFirstChild("Plots"); if not plots then return nil end
    local np, nd, nn = nil, math.huge, nil
    for _, plot in ipairs(plots:GetChildren()) do
        if isMyPlotByName(plot.Name) then continue end
        local podiums = plot:FindFirstChild("AnimalPodiums"); if not podiums then continue end
        for _, pod in ipairs(podiums:GetChildren()) do
            pcall(function()
                local base = pod:FindFirstChild("Base"); local spawn = base and base:FindFirstChild("Spawn")
                if spawn then local dist = (spawn.Position - h.Position).Magnitude; if dist < nd and dist <= Values.STEAL_RADIUS then local att = spawn:FindFirstChild("PromptAttachment"); if att then for _, ch in ipairs(att:GetChildren()) do if ch:IsA("ProximityPrompt") then np, nd, nn = ch, dist, pod.Name; break end end end end end
            end)
        end
    end
    return np, nd, nn
end

end
local function executeSteal(prompt, name)
    if isStealing then return end
    
    if not StealData[prompt] then
        StealData[prompt] = {hold = {}, trigger = {}, ready = true}
        pcall(function() if getconnections then for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do if c.Function then table.insert(StealData[prompt].hold, c.Function) end end; for _, c in ipairs(getconnections(prompt.Triggered)) do if c.Function then table.insert(StealData[prompt].trigger, c.Function) end end end end)
    end
    local data = StealData[prompt]; if not data.ready then return end; data.ready = false; isStealing = true; stealStartTime = tick()
    if ProgressLabel then ProgressLabel.Text = name or "Stealing..." end; if progressConnection then progressConnection:Disconnect() end
    progressConnection = RunService.Heartbeat:Connect(function()
        if not isStealing then progressConnection:Disconnect() return end; local prog = math.clamp((tick() - stealStartTime) / Values.STEAL_DURATION, 0, 1)
        if ProgressBarFill then ProgressBarFill.Size = UDim2.new(prog, 0, 1, 0) end; if ProgressPercentLabel then local percent = math.floor(prog * 100); ProgressPercentLabel.Text = getDiscordProgress(percent) end
    end)
    task.spawn(function()
        for _, f in ipairs(data.hold) do task.spawn(f) end
        task.wait(Values.STEAL_DURATION + math.random(1, 50)/1000)
        for _, f in ipairs(data.trigger) do task.spawn(f) end
        
        isStealSpeedLocked = true
        if not Enabled.SpeedWhileStealing then
            Enabled.SpeedWhileStealing = true
            if VisualSetters.SpeedWhileStealing then setVisual("SpeedWhileStealing", true) end
            if startSpeedWhileStealing then startSpeedWhileStealing() end
        end
        
        if Enabled.SpeedBoost then
            Enabled.SpeedBoost = false
            stopSpeedBoost()
            if VisualSetters.SpeedBoost then setVisual("SpeedBoost", false) end
        end
        
        if progressConnection then progressConnection:Disconnect() end; ResetProgressBar(); data.ready = true; isStealing = false
    end)
end

local function MakeSmartTouch(frame, onToggle, hasSettings)
    local dragging = false
    local dragInput = nil
    local dragStart = nil
    local startPos = nil
    local pressTime = 0
    
    local dragArea = Instance.new("Frame", frame)
    dragArea.Size = UDim2.new(1, hasSettings and (-30 * guiScale) or 0, 1, 0)
    dragArea.BackgroundTransparency = 1
    dragArea.ZIndex = 100
    
    dragArea.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            pressTime = tick()
            
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    
    dragArea.InputChanged:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            dragInput = input
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            local delta = input.Position - dragStart
            if (input.Position - dragStart).Magnitude > 5 then
                frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            end
        end
    end)
    
    dragArea.InputEnded:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
            dragging = false
            if tick() - pressTime < 0.3 and (input.Position - dragStart).Magnitude < 10 then
                if onToggle then onToggle() end
            end
        end
    end)
end

local function createButton(name, parent, enabledKey, toggleCallback, settingsFunc, defaultPos)
    if isMobile then
        name = name:gsub("%s*%[.*%]", "")
    end

    local isFloating = (parent == sg)
    local frame = Instance.new("Frame", parent)
    
    if isFloating then
        frame.AnchorPoint = Vector2.new(0.5, 0.5)
        frame.Size = UDim2.new(0, 70 * guiScale, 0, 70 * guiScale)
        if Values.ButtonPositions[enabledKey] then
            local p = Values.ButtonPositions[enabledKey]
            frame.Position = UDim2.new(p[1], p[2], p[3], p[4])
        else
            frame.Position = defaultPos
        end
        DraggableFrames[enabledKey] = frame
        DefaultPositions[enabledKey] = defaultPos
        
        local btnScale = Instance.new("UIScale", frame)
        btnScale.Scale = Values.Scale_Buttons
        table.insert(ButtonScales, btnScale)
    else
        frame.Size = UDim2.new(0.9, 0, 0, 35 * guiScale)
    end
    
    frame.BackgroundColor3 = C.bg
    frame.BackgroundTransparency = 0.15
    frame.BorderSizePixel = 0
    frame.Active = true
    frame.ClipsDescendants = true
    Instance.new("UICorner", frame).CornerRadius = UDim.new(0, 8)
    addGradient(frame)
    
    local frameStroke = Instance.new("UIStroke", frame)
    frameStroke.Color = Color3.fromRGB(0, 255, 0)
    frameStroke.Thickness = 1
    
    local label = Instance.new("TextLabel", frame)
    label.Size = UDim2.new(1, 0, 1, 0)
    label.BackgroundTransparency = 1
    label.Text = name
    label.TextColor3 = C.text
    label.Font = Enum.Font.GothamBold
    label.TextSize = isFloating and 9 * guiScale or 11 * guiScale
    label.TextXAlignment = Enum.TextXAlignment.Center
    label.TextYAlignment = Enum.TextYAlignment.Center
    
    local ind = Instance.new("Frame", label)
    ind.Size = UDim2.new(0, 8 * guiScale, 0, 8 * guiScale)
    local indOffset = settingsFunc and 35 or 15
    ind.Position = UDim2.new(1, -indOffset * guiScale, 1, -8 * guiScale)
    ind.BackgroundColor3 = Enabled[enabledKey] and C.on or C.off
    ind.BackgroundTransparency = 0
    Instance.new("UICorner", ind).CornerRadius = UDim.new(1, 0)
    
    local function updateVisual(state)
        if state then
            TweenService:Create(frame, TweenInfo.new(0.2), {BackgroundColor3 = C.activeBg}):Play()
            ind.BackgroundColor3 = C.on
            local glow = Instance.new("UIStroke", ind)
            glow.Name = "Glow"
            glow.Thickness = 1.5
            glow.Transparency = 0.4
            glow.Color = C.on
        else
            TweenService:Create(frame, TweenInfo.new(0.2), {BackgroundColor3 = C.bg}):Play()
            ind.BackgroundColor3 = C.off
            if ind:FindFirstChild("Glow") then ind:FindFirstChild("Glow"):Destroy() end
        end
    end
    
    if not VisualSetters[enabledKey] then VisualSetters[enabledKey] = {} end
    table.insert(VisualSetters[enabledKey], updateVisual)
    
    local function toggleAction()
        Enabled[enabledKey] = not Enabled[enabledKey]
        
        if enabledKey == "AutoWalkEnabled" then
            if Enabled[enabledKey] then
                if Enabled.AutoRightEnabled then
                    Enabled.AutoRightEnabled = false
                    if VisualSetters.AutoRightEnabled then setVisual("AutoRightEnabled", false) end
                    stopAutoRight()
                end
                if Enabled.BatAimbot then
                    Enabled.BatAimbot = false
                    if VisualSetters.BatAimbot then setVisual("BatAimbot", false) end
                    stopBatAimbot()
                end
                if not Enabled.SpeedBoost then
                    Enabled.SpeedBoost = true
                    startSpeedBoost()
                    if VisualSetters.SpeedBoost then setVisual("SpeedBoost", true) end
                end
                startAutoWalk()
            else
                stopAutoWalk()
            end
        elseif enabledKey == "AutoRightEnabled" then
            if Enabled[enabledKey] then
                if Enabled.AutoWalkEnabled then
                    Enabled.AutoWalkEnabled = false
                    if VisualSetters.AutoWalkEnabled then setVisual("AutoWalkEnabled", false) end
                    stopAutoWalk()
                end
                if Enabled.BatAimbot then
                    Enabled.BatAimbot = false
                    if VisualSetters.BatAimbot then setVisual("BatAimbot", false) end
                    stopBatAimbot()
                end
                if not Enabled.SpeedBoost then
                    Enabled.SpeedBoost = true
                    startSpeedBoost()
                    if VisualSetters.SpeedBoost then setVisual("SpeedBoost", true) end
                end
                startAutoRight()
            else
                stopAutoRight()
            end
        elseif enabledKey == "BatAimbot" then
            if Enabled[enabledKey] then
                if Enabled.AutoWalkEnabled then
                    Enabled.AutoWalkEnabled = false
                    if VisualSetters.AutoWalkEnabled then setVisual("AutoWalkEnabled", false) end
                    stopAutoWalk()
                end
                if Enabled.AutoRightEnabled then
                    Enabled.AutoRightEnabled = false
                    if VisualSetters.AutoRightEnabled then setVisual("AutoRightEnabled", false) end
                    stopAutoRight()
                end
                if not Enabled.SpeedBoost then
                    Enabled.SpeedBoost = true
                    startSpeedBoost()
                    if VisualSetters.SpeedBoost then setVisual("SpeedBoost", true) end
                end
                
                startBatAimbot()
            else
                stopBatAimbot()
            end
        elseif enabledKey == "InfJump" then
             if Enabled[enabledKey] then startInfJump() else stopInfJump() end
        elseif enabledKey == "HoldJump" then
            if Enabled[enabledKey] then
                startHoldJumpLoop()
                if Enabled.InfJump then
                    Enabled.InfJump = false
                    stopInfJump()
                    if VisualSetters.InfJump then setVisual("InfJump", false) end
                end
            else
                stopHoldJumpLoop()
            end
        elseif enabledKey == "ControllerMode" then
            if Enabled[enabledKey] then
                setupControllerBindings()
            else
                shutdownControllerBindings()
            end
        end
        
        if toggleCallback then toggleCallback(Enabled[enabledKey]) end
        setVisual(enabledKey, Enabled[enabledKey])
        playSound("rbxassetid://6895079813", 0.4, 1)
    end

    if isFloating then
        MakeSmartTouch(frame, toggleAction, settingsFunc ~= nil)
    else
        local btn = Instance.new("TextButton", frame)
        btn.Size = UDim2.new(1, settingsFunc and (-30 * guiScale) or 0, 1, 0)
        btn.BackgroundTransparency = 1
        btn.Text = ""
        btn.MouseButton1Click:Connect(toggleAction)
    end
    
    if settingsFunc then
        local gear = Instance.new("TextButton", frame)
        gear.Size = UDim2.new(0, 25 * guiScale, 0, 25 * guiScale)
        gear.Position = UDim2.new(1, -28 * guiScale, 1, -28 * guiScale)
        gear.BackgroundTransparency = 1
        gear.Text = "⚙"
        gear.TextColor3 = C.textDim
        gear.Font = Enum.Font.Gotham
        gear.TextSize = 12 * guiScale
        
        local expanded = false
        local settingsContainer = Instance.new("Frame", frame)
        settingsContainer.Size = UDim2.new(1, 0, 0, 0)
        settingsContainer.Position = UDim2.new(0, 0, 0, frame.Size.Y.Offset)
        settingsContainer.BackgroundTransparency = 1
        
        local height = settingsFunc(settingsContainer, label)
        local baseHeight = isFloating and 70 * guiScale or 35 * guiScale
        
        gear.MouseButton1Click:Connect(function()
            expanded = not expanded
            if expanded then
                frame.Size = UDim2.new(frame.Size.X.Scale, frame.Size.X.Offset, 0, baseHeight + (height * guiScale))
                gear.TextColor3 = C.accent
            else
                frame.Size = UDim2.new(frame.Size.X.Scale, frame.Size.X.Offset, 0, baseHeight)
                gear.TextColor3 = C.textDim
            end
        end)
    end
    
    return frame
end

local function createSliderInFrame(parent, label, min, max, valKey, yPos, callback)
    local l = Instance.new("TextLabel", parent)
    l.Size = UDim2.new(1, -10, 0, 14)
    l.Position = UDim2.new(0, 5, 0, yPos)
    l.BackgroundTransparency = 1
    l.Text = label
    l.TextColor3 = C.textDim
    l.Font = Enum.Font.Gotham
    l.TextSize = 9 * guiScale
    l.TextXAlignment = Enum.TextXAlignment.Center
    
    local valBox = Instance.new("TextBox", parent)
    valBox.Size = UDim2.new(0, 28, 0, 14)
    valBox.Position = UDim2.new(1, -32, 0, yPos)
    valBox.BackgroundColor3 = C.activeBg
    valBox.Text = tostring(Values[valKey])
    valBox.TextColor3 = C.muted
    valBox.Font = Enum.Font.GothamBold
    valBox.TextSize = 8 * guiScale
    Instance.new("UICorner", valBox).CornerRadius = UDim.new(1, 0)
    
    local sBg = Instance.new("Frame", parent)
    sBg.Size = UDim2.new(0.9, 0, 0, 5)
    sBg.Position = UDim2.new(0.05, 0, 0, yPos + 20)
    sBg.BackgroundColor3 = C.activeBg
    Instance.new("UICorner", sBg).CornerRadius = UDim.new(1, 0)
    
    local fill = Instance.new("Frame", sBg)
    fill.Size = UDim2.new((Values[valKey] - min)/(max - min), 0, 1, 0)
    fill.BackgroundColor3 = C.accent
    Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)
    
    local btn = Instance.new("TextButton", sBg)
    btn.Size = UDim2.new(1, 0, 3, 0)
    btn.Position = UDim2.new(0, 0, -1, 0)
    btn.BackgroundTransparency = 1
    btn.Text = ""
    
    local function update(input)
        local pos = math.clamp((input.Position.X - sBg.AbsolutePosition.X) / sBg.AbsoluteSize.X, 0, 1)
        local newVal = math.floor((min + (max-min)*pos) * 10) / 10
        Values[valKey] = newVal
        valBox.Text = tostring(newVal)
        fill.Size = UDim2.new(pos, 0, 1, 0)
        if callback then callback(newVal) end
    end
    
    local dragging = false
    btn.MouseButton1Down:Connect(function() dragging = true end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = false end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then update(input) end
    end)
    
    valBox.FocusLost:Connect(function()
        local n = tonumber(valBox.Text)
        if n then
            n = math.clamp(n, min, max)
            Values[valKey] = n; valBox.Text = tostring(n); fill.Size = UDim2.new((n-min)/(max-min), 0, 1, 0)
            if callback then callback(n) end
        end
    end)
    
    SliderSetters[valKey] = function(v)
        local n = math.clamp(v, min, max)
        Values[valKey] = n; valBox.Text = tostring(n); fill.Size = UDim2.new((n-min)/(max-min), 0, 1, 0)
    end
    
    return 28
end

    if Enabled.BatAimbot then startBatAimbot() end

end)

-- Keybind handling (PC only, controller uses ContextActionService)
UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe or IsBinding then return end
    if input.KeyCode == Enum.KeyCode.U then sg.Enabled = not sg.Enabled return end
    
    -- Skip controller bindings if controller mode is enabled (they're handled by ContextActionService)
    if Enabled.ControllerMode then
        -- Still allow PC keybinds for the UI toggle (U)
        return
    end
    
    if input.KeyCode == KEYBINDS.SPEED then
        Enabled.SpeedBoost = not Enabled.SpeedBoost
        setVisual("SpeedBoost", Enabled.SpeedBoost)
        if Enabled.SpeedBoost then startSpeedBoost() else stopSpeedBoost() end
    end
    
    if input.KeyCode == KEYBINDS.BATAIMBOT then
        Enabled.BatAimbot = not Enabled.BatAimbot
        if Enabled.BatAimbot then
            Enabled.AutoWalkEnabled = false
            Enabled.AutoRightEnabled = false
            setVisual("AutoWalkEnabled", false)
            setVisual("AutoRightEnabled", false)
            
            if not Enabled.SpeedBoost then
                Enabled.SpeedBoost = true
                startSpeedBoost()
                setVisual("SpeedBoost", true)
            end

            stopAutoWalk()
            stopAutoRight()
            startBatAimbot()
        else
            stopBatAimbot()
        end
        setVisual("BatAimbot", Enabled.BatAimbot)
    end
    
    if input.KeyCode == KEYBINDS.AUTOLEFT then
        Enabled.AutoWalkEnabled = not Enabled.AutoWalkEnabled
        if Enabled.AutoWalkEnabled then
            Enabled.AutoRightEnabled = false
            Enabled.BatAimbot = false
            setVisual("AutoRightEnabled", false)
            setVisual("BatAimbot", false)
            
            if not Enabled.SpeedBoost then
                Enabled.SpeedBoost = true
                startSpeedBoost()
                setVisual("SpeedBoost", true)
            end

            stopAutoRight()
            stopBatAimbot()
            startAutoWalk()
        else
            stopAutoWalk()
        end
        setVisual("AutoWalkEnabled", Enabled.AutoWalkEnabled)
    end
    
    if input.KeyCode == KEYBINDS.AUTORIGHT then
        Enabled.AutoRightEnabled = not Enabled.AutoRightEnabled
        if Enabled.AutoRightEnabled then
            Enabled.AutoWalkEnabled = false
            Enabled.BatAimbot = false
            setVisual("AutoWalkEnabled", false)
            setVisual("BatAimbot", false)
            
            if not Enabled.SpeedBoost then
                Enabled.SpeedBoost = true
                startSpeedBoost()
                setVisual("SpeedBoost", true)
            end

            stopAutoWalk()
            stopBatAimbot()
            startAutoRight()
        else
            stopAutoRight()
        end
        setVisual("AutoRightEnabled", Enabled.AutoRightEnabled)
    end
    
    if input.KeyCode == KEYBINDS.TPDOWN then
        performTPDown()
    end
    
    if input.KeyCode == KEYBINDS.HOLDJUMP then
        Enabled.HoldJump = not Enabled.HoldJump
        setVisual("HoldJump", Enabled.HoldJump)
        if Enabled.HoldJump then
            if Enabled.InfJump then
                Enabled.InfJump = false
                stopInfJump()
                if VisualSetters.InfJump then setVisual("InfJump", false) end
            end
            startHoldJumpLoop()
        else
            stopHoldJumpLoop()
        end
    end
end)
