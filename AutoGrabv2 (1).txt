local desyncBtn = makeBtn("ACTIVATE", 90)

-- UNWALK LOGIC (Restores/Destroys Animate script)
local function startUnwalk()
    local c = player.Character;
    if not c then return end
    local hum = c:FindFirstChildOfClass("Humanoid");
    if hum then
        for _, t in ipairs(hum:GetPlayingAnimationTracks()) do t:Stop() end
        hum.WalkSpeed = 0
        hum.JumpPower = 0
    end
    local anim = c:FindFirstChild("Animate");
    if anim then
        G.unwalkSavedAnimate = anim:Clone();
        anim:Destroy()
    end
end

local function stopUnwalk()
    local c = player.Character;
    if c then
        local hum = c:FindFirstChildOfClass("Humanoid")
        if hum then hum.WalkSpeed = 16 hum.JumpPower = 50 end
        if G.unwalkSavedAnimate then
            G.unwalkSavedAnimate:Clone().Parent = c;
            G.unwalkSavedAnimate = nil
        end
    end
end

-- MAIN TOGGLES
local desyncEnabled = false
local unwalkEnabled = false

desyncBtn.MouseButton1Click:Connect(function()
    desyncEnabled = not desyncEnabled
    
    if desyncEnabled then
        -- ENABLE DESYNC (raknet source)
        pcall(function()
            if raknet then raknet.desync(true) end
            local character = player.Character
            if character and character:FindFirstChild("Humanoid") then
                character.Humanoid.Health = 0 -- Required respawn logic
                task.wait(0.5)
            end
        end)
        desyncBtn.Text = "DEACTIVATE"
        status.Text = "Active"
        status.TextColor3 = Color3.fromRGB(255, 255, 255)
    else
        -- DISABLE DESYNC
        pcall(function()
            if raknet then raknet.desync(false) end
        end)
        desyncBtn.Text = "ACTIVATE"
        status.Text = "Inactive"
        status.TextColor3 = Color3.fromRGB(65, 65, 70)
    end
end)

unwalkBtn.MouseButton1Click:Connect(function()
    unwalkEnabled = not unwalkEnabled
    unwalkBtn.Text = unwalkEnabled and "UNWALK: ON" or "UNWALK: OFF"
    if unwalkEnabled then startUnwalk() else stopUnwalk() end
end)
